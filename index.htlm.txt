    <script>
        const videoModule = {
            currentIndex: 0,
            channelsPerPage: 6,
            streams: [],
            players: [],

            extractStreams: function(rawData) {
                this.streams = [];
                rawData.forEach(row => {
                    Object.keys(row).forEach(key => {
                        const val = row[key];
                        if (typeof val === 'string' && (val.includes('.m3u8') || val.startsWith('http'))) {
                            this.streams.push({
                                name: `[${row['País'] || 'LATAM'}] ${row['Nombre del Proyecto'] || key}`,
                                url: val
                            });
                        }
                    });
                });
            },

            renderPanel: function() {
                const grid = document.getElementById('video-grid');
                const info = document.getElementById('mosaic-page-info');
                
                this.players.forEach(p => { if(p) p.destroy(); });
                this.players = [];
                grid.innerHTML = '';
                
                if (this.streams.length === 0) {
                    grid.innerHTML = '<div style="grid-column: 1/-1; padding: 20px; text-align: center; color: var(--text-dim);">No se detectaron flujos de video (.m3u8) en la hoja de datos actual.</div>';
                    info.innerText = 'Panel 0 de 0';
                    return;
                }

                const totalPages = Math.ceil(this.streams.length / this.channelsPerPage);
                const currentPage = Math.floor(this.currentIndex / this.channelsPerPage) + 1;
                info.innerText = `Panel ${currentPage} de ${totalPages}`;

                const pageItems = this.streams.slice(this.currentIndex, this.currentIndex + this.channelsPerPage);

                pageItems.forEach((stream, i) => {
                    const id = `video-player-${i}`;
                    grid.innerHTML += `
                        <div class="video-slot">
                            <video id="${id}" controls muted autoplay></video>
                            <div class="video-label">${stream.name}</div>
                        </div>
                    `;
                });

                setTimeout(() => {
                    pageItems.forEach((stream, i) => {
                        const video = document.getElementById(`video-player-${i}`);
                        if (Hls.isSupported()) {
                            const hls = new Hls({ enableWorker: true, lowLatencyMode: true });
                            hls.loadSource(stream.url);
                            hls.attachMedia(video);
                            this.players.push(hls);
                        } else if (video.canPlayType('application/vnd.apple.mpegurl')) {
                            video.src = stream.url;
                        }
                    });
                }, 100);
            },

            nextPanel: function() {
                if (this.currentIndex + this.channelsPerPage < this.streams.length) {
                    this.currentIndex += this.channelsPerPage;
                    this.renderPanel();
                }
            },

            prevPanel: function() {
                if (this.currentIndex - this.channelsPerPage >= 0) {
                    this.currentIndex -= this.channelsPerPage;
                    this.renderPanel();
                }
            }
        };

        const app = {
            state: {
                mode: 'ceo',
                activeTab: 'projects',
                data: [],
                filters: { region: 'all', country: 'all', status: 'all' }
            },

            schemaMap: {
                id: ['ID UFINET', 'ID', 'IdUfinet', 'ProjectID'],
                pais: ['País', 'Country', 'PAIS', 'Ubicación'],
                nombre: ['Nombre del Proyecto', 'Proyecto', 'Name'],
                avance_real: ['Avance Real', 'Real %', '% Real'],
                avance_plan: ['Avance Planificado', 'Planned %', '% Plan'],
                costo_real: ['Costo Real', 'Real Cost', 'Gasto'],
                costo_plan: ['Costo Planificado', 'Budget'],
                estado: ['Estado', 'Status', 'Etapa']
            },

            init: function() {
                const url = document.getElementById('sheet-url').value;
                if (!url) {
                    this.loadDemoData();
                    this.startup();
                } else {
                    Papa.parse(url, {
                        download: true,
                        header: true,
                        complete: (results) => {
                            this.processRawData(results.data);
                            this.startup();
                        },
                        error: (err) => {
                            console.error("Error cargando CSV:", err);
                            alert("No se pudo cargar el documento. Verifica el enlace CSV.");
                            this.loadDemoData();
                            this.startup();
                        }
                    });
                }
            },

            processRawData: function(rawData) {
                this.state.data = rawData.map(row => {
                    let n = { ...row }; 
                    for (let key in this.schemaMap) {
                        let alias = this.schemaMap[key].find(a => row.hasOwnProperty(a));
                        n[key] = alias ? row[alias] : null;
                    }
                    n.avance_real = parseFloat(n.avance_real) || 0;
                    n.avance_plan = parseFloat(n.avance_plan) || 0;
                    n.costo_real = parseFloat(n.costo_real) || 0;
                    n.costo_plan = parseFloat(n.costo_plan) || 0;
                    return n;
                });
                videoModule.extractStreams(this.state.data);
            },

            startup: function() {
                this.bindEvents();
                this.renderAll();
                lucide.createIcons();
            },

            bindEvents: function() {
                document.querySelectorAll('.mode-btn[data-mode]').forEach(btn => {
                    btn.onclick = (e) => {
                        document.querySelectorAll('.mode-btn[data-mode]').forEach(b => b.classList.remove('active'));
                        e.currentTarget.classList.add('active');
                        this.state.mode = e.currentTarget.dataset.mode;
                        document.getElementById('view-title').innerText = this.state.mode === 'ceo' ? 'Dashboard Ejecutivo CEO' : 'Control Operativo Detallado';
                        document.getElementById('charts-section').style.display = this.state.mode === 'ceo' ? 'grid' : 'none';
                        this.renderAll();
                    };
                });

                ['country', 'status'].forEach(f => {
                    const el = document.getElementById(`filter-${f}`);
                    el.onchange = (e) => {
                        this.state.filters[f] = e.target.value;
                        this.renderAll();
                    };
                });
            },

            switchTab: function(tabId) {
                this.state.activeTab = tabId;
                document.querySelectorAll('.mode-btn[data-tab]').forEach(b => b.classList.remove('active'));
                document.querySelector(`.mode-btn[data-tab="${tabId}"]`).classList.add('active');
                
                const mosaicSec = document.getElementById('mosaic-section');
                if(tabId === 'channels') {
                    mosaicSec.classList.remove('hidden');
                    videoModule.renderPanel();
                } else {
                    mosaicSec.classList.add('hidden');
                    videoModule.players.forEach(p => { if(p) p.destroy(); });
                }
                
                this.renderTable();
                lucide.createIcons();
            },

            renderAll: function() {
                this.renderKPIs();
                if(this.state.mode === 'ceo') this.renderCharts();
                this.renderTable();
                this.updateFilterOptions();
            },

            updateFilterOptions: function() {
                const countries = [...new Set(this.state.data.map(r => r.pais).filter(Boolean))];
                const states = [...new Set(this.state.data.map(r => r.estado).filter(Boolean))];
                
                this.populateSelect('filter-country', countries, 'Todos');
                this.populateSelect('filter-status', states, 'Todos');
            },

            populateSelect: function(id, items, defaultText) {
                const el = document.getElementById(id);
                const currentVal = el.value;
                el.innerHTML = `<option value="all">${defaultText}</option>`;
                items.forEach(item => {
                    const opt = document.createElement('option');
                    opt.value = item;
                    opt.innerText = item;
                    el.appendChild(opt);
                });
                el.value = currentVal;
            },

            renderKPIs: function() {
                const filtered = this.state.data.filter(r => 
                    (this.state.filters.country === 'all' || r.pais === this.state.filters.country) &&
                    (this.state.filters.status === 'all' || r.estado === this.state.filters.status)
                );
                
                const total = filtered.length;
                const avgProg = total ? filtered.reduce((a, r) => a + r.avance_real, 0) / total : 0;
                const risks = filtered.filter(r => r.avance_real < (r.avance_plan * 0.8)).length;

                document.getElementById('kpi-section').innerHTML = `
                    <div class="card">
                        <div class="kpi-title"><i data-lucide="layers"></i> Proyectos</div>
                        <div class="kpi-value">${total}</div>
                        <div class="kpi-trend success">Activos en Selección</div>
                    </div>
                    <div class="card">
                        <div class="kpi-title"><i data-lucide="activity"></i> Avance Medio</div>
                        <div class="kpi-value">${avgProg.toFixed(1)}%</div>
                        <div class="kpi-trend ${avgProg >= 80 ? 'success' : 'warning'}">Global Real</div>
                    </div>
                    <div class="card">
                        <div class="kpi-title"><i data-lucide="alert-triangle"></i> En Riesgo</div>
                        <div class="kpi-value ${risks > 0 ? 'danger' : ''}">${risks}</div>
                        <div class="kpi-trend">Atraso Crítico</div>
                    </div>
                `;
            },

            renderCharts: function() {
                const ctx = document.getElementById('chart-trends').getContext('2d');
                if(window.myChart) window.myChart.destroy();
                window.myChart = new Chart(ctx, {
                    type: 'line',
                    data: {
                        labels: ['Ene', 'Feb', 'Mar', 'Abr', 'May', 'Jun'],
                        datasets: [
                            { label: 'Real', data: [10, 25, 45, 60, 75, 88], borderColor: '#38bdf8', tension: 0.4 },
                            { label: 'Plan', data: [15, 30, 50, 65, 80, 95], borderColor: '#94a3b8', borderDash: [5,5], tension: 0.4 }
                        ]
                    },
                    options: { responsive: true, maintainAspectRatio: false }
                });

                const dataHeat = [{ z: [[10, 20], [20, 1], [30, 60]], x: ['GUA', 'SLV'], y: ['Construcción', 'Core', 'Ventas'], type: 'heatmap', colorscale: 'Viridis' }];
                Plotly.newPlot('chart-heatmap', dataHeat, { paper_bgcolor: 'transparent', plot_bgcolor: 'transparent', font: {color: '#94a3b8'}, margin: { t: 10, b: 30, l: 80, r: 10 } });
            },

            renderTable: function() {
                const container = document.getElementById('table-container');
                const filtered = this.state.data.filter(r => 
                    (this.state.filters.country === 'all' || r.pais === this.state.filters.country) &&
                    (this.state.filters.status === 'all' || r.estado === this.state.filters.status)
                );

                let html = `<table><thead><tr>`;
                
                if(this.state.activeTab === 'projects') {
                    html += `<th>ID UFINET</th><th>Proyecto</th><th>País</th><th>Estado</th><th>Avance</th><th>Salud</th></tr></thead><tbody>`;
                    filtered.slice(0, 50).forEach(r => {
                        const sem = r.avance_real >= r.avance_plan ? 'success' : (r.avance_real < r.avance_plan * 0.7 ? 'danger' : 'warning');
                        html += `<tr>
                            <td><b>${r.id || '-'}</b></td><td>${r.nombre || 'N/A'}</td><td>${r.pais || '-'}</td>
                            <td><span class="status-badge" style="background: var(--border)">${r.estado || 'S/D'}</span></td>
                            <td>${r.avance_real}%</td>
                            <td><div style="width:12px; height:12px; border-radius:50%; background: var(--${sem})"></div></td>
                        </tr>`;
                    });
                } else if(this.state.activeTab === 'inventory') {
                    html += `<th>País</th><th>ID Equipo</th><th>Tipo Activo</th><th>Capacidad</th><th>Estado</th></tr></thead><tbody>`;
                    filtered.forEach(r => {
                        html += `<tr><td>${r.pais}</td><td>${r.id || 'N/A'}</td><td>Core Node</td><td>85%</td><td><span class="success">Óptimo</span></td></tr>`;
                    });
                } else {
                    html += `<th>País</th><th>Canal / Proyecto</th><th>Endpoint</th><th>Estado</th></tr></thead><tbody>`;
                    filtered.forEach(r => {
                        html += `<tr><td>${r.pais}</td><td>${r.nombre || r.id}</td><td style="font-family:monospace; font-size:0.7rem;">Detectado en Columnas Clientes</td><td><span class="success">Activo</span></td></tr>`;
                    });
                }
                html += `</tbody></table>`;
                container.innerHTML = html;
            },

            loadDemoData: function() {
                const countries = ['Guatemala', 'El Salvador', 'Honduras'];
                const demo = [];
                for(let i=0; i<5; i++) {
                    demo.push({
                        'ID UFINET': 'DEMO-' + i,
                        'Nombre del Proyecto': 'Sitio Demostración ' + i,
                        'País': countries[i % 3],
                        'Estado': 'Construcción',
                        'Avance Real': 50,
                        'Avance Planificado': 70
                    });
                }
                this.processRawData(demo);
            }
        };

        window.onload = () => app.init();
    </script>
